package tourGuide.service;

import gpsUtil.location.Attraction;
import gpsUtil.location.Location;

import java.util.Comparator;

public abstract class SortAttractionsByDistance implements Comparator<Attraction>  {

    private final RewardsService rewardsService;

    public SortAttractionsByDistance(RewardsService rewardsService) {
        this.rewardsService = rewardsService;
    }

    public double compare(Attraction a, Attraction b, Location location)
    {
        return rewardsService.getDistance(a, location) - rewardsService.getDistance(b, location);
    }
}
